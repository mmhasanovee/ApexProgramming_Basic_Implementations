public class TaskUtility {
    
    public static String getTaskPriority(String leadCountry){
        
        
        if(String.isBlank(leadCountry)||leadCountry.length()>2){
            
            return null;
        }
        
        String taskPriority;
        
        if(leadCountry=='US'){
            
            taskPriority='High';
        }
        
        else{
            
            taskPriority='Normal';
        }
        
        return taskPriority;
    }

}
